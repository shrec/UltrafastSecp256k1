145// ============================================================================
146// Montgomery Domain Constants
147// ============================================================================
148// Montgomery domain: R = 2^256 mod p
149// For secp256k1, p = 2^256 - 0x1000003D1, so R = 0x1000003D1
150//
151// Usage:
152// - to_mont(a) = (a * R²) mod p // Convert a → a*R
153// - from_mont(aR) = (aR * R⁻¹) mod p // Convert a*R → a
154// - mont_mul(aR, bR) = (aR * bR * R⁻¹) mod p = (ab)R // Stays in Montgomery
155//
156// Benefits:
157// - Modular reduction uses cheap multiplication instead of division
158// - 10-15% faster in batch operations
159// - Required for GPU H-based inversion pipeline
160
161namespace montgomery {
162 // R = 2^256 mod p = 0x1000003D1
163 inline const FieldElement& R() {
164 static const FieldElement r = FieldElement::from_uint64(0x1000003D1ULL);
165 return r;
166 }
167
168 // R² mod p = (2^256)² mod p = 0x000007A2000E90A1
169 inline const FieldElement& R2() {
170 static const FieldElement r2 = FieldElement::from_limbs(
171 {0x000007A2000E90A1ULL, 0x0000000000000001ULL, 0ULL, 0ULL});
172 return r2;
173 }
174
175 // R³ mod p = (2^256)³ mod p
176 inline const FieldElement& R3() {
177 static const FieldElement r3 = FieldElement::from_limbs(
178 {0x002BB1E33795F671ULL, 0x0000000100000B73ULL, 0ULL, 0ULL});
179 return r3;
180 }
181
182 // R⁻¹ mod p = (2^256)⁻¹ mod p
183 inline const FieldElement& R_inv() {
184 static const FieldElement r_inv = FieldElement::from_limbs(
185 {0xD838091D0868192AULL, 0xBCB223FEDC24A059ULL,
186 0x9C46C2C295F2B761ULL, 0xC9BD190515538399ULL});
187 return r_inv;
188 }

191 constexpr std::uint64_t K_MOD = 0x1000003D1ULL;
192}
