109// 64x64 → 128-bit multiplication using MULX
110// MULX: multiplicand in RDX, result in two registers (no flag updates!)
111inline void mulx64(uint64_t a, uint64_t b, uint64_t& lo, uint64_t& hi) {
112 #if defined(_MSC_VER)
113 // MSVC intrinsic
114 lo = _mulx_u64(a, b, &hi);
115 #elif defined(__GNUC__) || defined(__clang__)
116 #if defined(__BMI2__)
117 // GCC/Clang intrinsic with BMI2
118 lo = _mulx_u64(a, b, (unsigned long long*)&hi);
119 #else
120 // GCC/Clang fallback
121 #ifdef SECP256K1_NO_INT128
122 // 32-bit safe implementation
123 uint64_t a_lo = a & 0xFFFFFFFFULL;
124 uint64_t a_hi = a >> 32;
125 uint64_t b_lo = b & 0xFFFFFFFFULL;
126 uint64_t b_hi = b >> 32;
127
128 uint64_t p0 = a_lo * b_lo;
129 uint64_t p1 = a_lo * b_hi;
130 uint64_t p2 = a_hi * b_lo;
131 uint64_t p3 = a_hi * b_hi;
132
133 uint64_t carry = ((p0 >> 32) + (p1 & 0xFFFFFFFFULL) + (p2 & 0xFFFFFFFFULL)) >> 32;
134
135 lo = p0 + (p1 << 32) + (p2 << 32);
136 hi = p3 + (p1 >> 32) + (p2 >> 32) + carry;
137 #else
138 __uint128_t result = static_cast<__uint128_t>(a) * b;
139 lo = static_cast<uint64_t>(result);
140 hi = static_cast<uint64_t>(result >> 64);
141 #endif
142 #endif
143 #else
144 // Fallback
145 // #ifdef SECP256K1_NO_INT128
146 // // 32-bit safe implementation
147 // uint64_t a_lo = a & 0xFFFFFFFFULL;
148 // uint64_t a_hi = a >> 32;
149 // uint64_t b_lo = b & 0xFFFFFFFFULL;
150 // uint64_t b_hi = b >> 32;
151 //
152 // uint64_t p0 = a_lo * b_lo;
153 // uint64_t p1 = a_lo * b_hi;
154 // uint64_t p2 = a_hi * b_lo;
155 // uint64_t p3 = a_hi * b_hi;
156 //
157 // uint64_t carry = ((p0 >> 32) + (p1 & 0xFFFFFFFFULL) + (p2 & 0xFFFFFFFFULL)) >> 32;
158 //
159 // lo = p0 + (p1 << 32) + (p2 << 32);
160 // hi = p3 + (p1 >> 32) + (p2 >> 32) + carry;
161 // #else
162 // __uint128_t result = static_cast<__uint128_t>(a) * b;
163 // lo = static_cast<uint64_t>(result);
164 // hi = static_cast<uint64_t>(result >> 64);
165 // #endif
166 #endif
167}

169// Add with carry using ADCX (carry flag chain)
170// x86-only: BMI2/ADX intrinsic
171#if defined(__x86_64__) || defined(_M_X64)
172inline uint8_t adcx64(uint64_t a, uint64_t b, uint8_t carry, uint64_t& result) {
173 #if defined(_MSC_VER)
174 // MSVC intrinsic
175 return _addcarry_u64(carry, a, b, reinterpret_cast<unsigned long long*>(&result));
176 #else
177 // GCC/Clang intrinsic (_addcarry_u64 in x86intrin.h/immintrin.h)
178 return _addcarry_u64(carry, a, b, (unsigned long long*)&result);
179 #endif
180}
181#else
182// Portable fallback for non-x86 (RISC-V, ARM, ESP32, etc.)
183inline uint8_t adcx64(uint64_t a, uint64_t b, uint8_t carry, uint64_t& result) {
184 #ifdef SECP256K1_NO_INT128
185 // 32-bit safe implementation
186 result = a + b;
187 uint8_t new_carry = (result < a) ? 1 : 0;
188 if (carry) {
189 uint64_t temp = result + 1;
190 new_carry |= (temp < result) ? 1 : 0;
191 result = temp;
192 }
193 return new_carry;
194 #else
195 unsigned __int128 sum = static_cast<unsigned __int128>(a) +
196 static_cast<unsigned __int128>(b) +
197 static_cast<unsigned __int128>(carry);
198 result = static_cast<uint64_t>(sum);
199 return static_cast<uint8_t>(sum >> 64);
200 #endif
201}
202#endif

204// Add with overflow using ADOX (overflow flag chain, independent of ADCX!)
205inline uint8_t adox64(uint64_t a, uint64_t b, uint8_t overflow, uint64_t& result) {
206 // Note: ADOX not directly available as intrinsic, use ADCX for now
207 // In inline assembly, we'd use separate flag chains
208 return adcx64(a, b, overflow, result);
