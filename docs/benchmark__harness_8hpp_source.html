115struct Timer {
116 // Returns a monotonic tick count.
117 // On x86: RDTSC cycles (serialized with RDTSCP or CPUID+RDTSC).
118 // Elsewhere: nanoseconds from chrono.
119 static inline uint64_t now() noexcept {
120#if BENCH_HAS_RDTSC
121# if defined(_MSC_VER)
122 unsigned int aux;
123 return __rdtscp(&aux);
124# else
125 uint32_t lo, hi;
126 // RDTSCP serializes instruction stream and reads TSC
127 asm volatile("rdtscp" : "=a"(lo), "=d"(hi) : : "%ecx");
128 return (static_cast<uint64_t>(hi) << 32) | lo;
129# endif
130#else
131 return static_cast<uint64_t>(
132 std::chrono::high_resolution_clock::now().time_since_epoch().count());
133#endif
134 }
135
136 // Ticks -> nanoseconds conversion.
137 // On x86 with RDTSC: calibrates TSC frequency at first call.
138 // On non-x86: ticks ARE nanoseconds (chrono).
139 static double ticks_to_ns(uint64_t ticks) noexcept {
140#if BENCH_HAS_RDTSC
141 static const double ns_per_tick = calibrate_tsc();
142 return static_cast<double>(ticks) * ns_per_tick;
143#else
144 // chrono ticks -- convert to nanoseconds using the clock's period
145 using Period = std::chrono::high_resolution_clock::period;
146 constexpr double ratio = static_cast<double>(Period::num) / static_cast<double>(Period::den) * 1e9;
147 return static_cast<double>(ticks) * ratio;
148#endif
149 }
150
151 // Returns the timer type name for display
152 static const char* timer_name() noexcept {
153#if BENCH_HAS_RDTSC
154 return "RDTSCP";
155#else
156 return "chrono::high_resolution_clock";
157#endif
158 }
159
160private:
161#if BENCH_HAS_RDTSC
162 // Calibrate TSC: measure ~10ms with chrono to get ns/tick ratio
163 static double calibrate_tsc() noexcept {
164 using Clock = std::chrono::high_resolution_clock;
165 constexpr int CALIBRATE_MS = 10;
166
167 // Warm up
168 volatile uint64_t warm = now();
169 (void)warm;
170
171 auto chrono_start = Clock::now();
172 uint64_t tsc_start = now();
173
174 // Spin for ~10ms
175 auto target = chrono_start + std::chrono::milliseconds(CALIBRATE_MS);
176 while (Clock::now() < target) {
177 // busy-wait
178 }
179
180 uint64_t tsc_end = now();
181 auto chrono_end = Clock::now();
182
183 double ns_elapsed = static_cast<double>(
184 std::chrono::duration_cast<std::chrono::nanoseconds>(chrono_end - chrono_start).count());
185 double tsc_elapsed = static_cast<double>(tsc_end - tsc_start);
186
187 return (tsc_elapsed > 0.0) ? (ns_elapsed / tsc_elapsed) : 1.0;
188 }
189#endif
190};

194struct Stats {
195 double min_ns = 0.0;
196 double max_ns = 0.0;
197 double median_ns = 0.0;
198 double mean_ns = 0.0;
199 double stddev_ns = 0.0;
200 int samples = 0; // after outlier removal
201 int outliers = 0; // removed samples
202};
203
204// Compute stats with IQR outlier removal
205inline Stats compute_stats(std::vector<double>& data) {
206 Stats s{};
207 if (data.empty()) return s;
208
209 std::sort(data.begin(), data.end());
210
211 const std::size_t n = data.size();
212
213 if (n < 4) {
214 // Too few samples for IQR -- use all
215 s.min_ns = data.front();
216 s.max_ns = data.back();
217 s.median_ns = data[n / 2];
218 double sum = 0.0;
219 for (auto v : data) sum += v;
220 s.mean_ns = sum / static_cast<double>(n);
221 s.samples = static_cast<int>(n);
222 s.outliers = 0;
223 return s;
224 }
225
226 // IQR-based outlier removal
227 double q1 = data[n / 4];
228 double q3 = data[(3 * n) / 4];
229 double iqr = q3 - q1;
230 double lower = q1 - 1.5 * iqr;
231 double upper = q3 + 1.5 * iqr;
232
233 std::vector<double> filtered;
234 filtered.reserve(n);
235 for (auto v : data) {
236 if (v >= lower && v <= upper) {
237 filtered.push_back(v);
238 }
239 }
240
241 if (filtered.empty()) {
242 // All outliers? Fall back to raw data
243 filtered = data;
244 }
245
246 const std::size_t fn = filtered.size();
247 s.outliers = static_cast<int>(n - fn);
248 s.samples = static_cast<int>(fn);
249 s.min_ns = filtered.front();
250 s.max_ns = filtered.back();
251 s.median_ns = filtered[fn / 2];
252
253 double sum = 0.0;
254 for (auto v : filtered) sum += v;
255 s.mean_ns = sum / static_cast<double>(fn);
256
257 double var = 0.0;
258 for (auto v : filtered) {
259 double d = v - s.mean_ns;
260 var += d * d;
261 }
262 s.stddev_ns = std::sqrt(var / static_cast<double>(fn));
263
264 return s;
265}

285class Harness {
286public:
287 int warmup_iters = 500; // warmup iterations per function call
288 std::size_t passes = 11; // measurement passes (odd number for clean median)
289
290 Harness() = default;
291 Harness(int warmup, std::size_t p) : warmup_iters(warmup), passes(p) {}
292
293 // Run benchmark: returns median ns/iter after IQR outlier removal.
294 // Func is called `iters` times per pass.
295 template <typename Func>
296 double run(int iters, Func&& func) const {
297 // Warmup
298 for (int i = 0; i < warmup_iters; ++i) {
299 func();
300 ClobberMemory();
301 }
302
303 // Measurement passes
304 std::vector<double> ns_per_iter;
305 ns_per_iter.reserve(passes);
306
307 for (std::size_t p = 0; p < passes; ++p) {
308 uint64_t t0 = Timer::now();
309 for (int i = 0; i < iters; ++i) {
310 func();
311 }
312 ClobberMemory();
313 uint64_t t1 = Timer::now();
314
315 double total_ns = Timer::ticks_to_ns(t1 - t0);
316 ns_per_iter.push_back(total_ns / iters);
317 }
318
319 Stats st = compute_stats(ns_per_iter);
320 return st.median_ns;
321 }
322

351 // Convenience: run + print one-liner result
352 template <typename Func>
353 double run_and_print(const char* name, int iters, Func&& func) const {
354 Stats st = run_stats(iters, func);
355 std::printf(" %-28s %9.2f ns (min=%6.2f median=%6.2f stddev=%5.2f n=%d-%d)\n",
356 name, st.median_ns, st.min_ns, st.median_ns, st.stddev_ns,
357 st.samples, st.outliers);
