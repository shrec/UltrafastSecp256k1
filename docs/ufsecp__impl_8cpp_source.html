86static inline Point point_from_compressed(const uint8_t pub[33]) {
87 std::array<uint8_t, 32> x_bytes;
88 std::memcpy(x_bytes.data(), pub + 1, 32);
89 auto x = FE::from_bytes(x_bytes);
90
91 /* y² = x³ + 7 */
92 auto x2 = x * x;
93 auto x3 = x2 * x;
94 auto y2 = x3 + FE::from_uint64(7);
95
96 /* sqrt via addition chain for (p+1)/4 */
97 auto t = y2;
98 auto a = t.square() * t;
99 auto b = a.square() * t;
100 auto c = b.square().square().square() * b;
101 auto d = c.square().square().square() * b;
102 auto e = d.square().square() * a;
103 auto f = e;
104 for (int i = 0; i < 11; ++i) f = f.square();
105 f = f * e;
106 auto g = f;
107 for (int i = 0; i < 22; ++i) g = g.square();
108 g = g * f;
109 auto h = g;
110 for (int i = 0; i < 44; ++i) h = h.square();
111 h = h * g;
112 auto j = h;
113 for (int i = 0; i < 88; ++i) j = j.square();
114 j = j * h;
115 auto k = j;
116 for (int i = 0; i < 44; ++i) k = k.square();
117 k = k * g;
118 auto m = k.square().square().square() * b;
119 auto y = m;
120 for (int i = 0; i < 23; ++i) y = y.square();
121 y = y * f;
122 for (int i = 0; i < 6; ++i) y = y.square();
123 y = y * a;
124 y = y.square().square();
125
126 auto y_bytes = y.to_bytes();
127 bool y_is_odd = (y_bytes[31] & 1) != 0;
128 bool want_odd = (pub[0] == 0x03);
129 if (y_is_odd != want_odd)
130 y = FE::from_uint64(0) - y;
131
132 return Point::from_affine(x, y);
133}

182ufsecp_error_t ufsecp_ctx_create(ufsecp_ctx** ctx_out) {
183 if (!ctx_out) return UFSECP_ERR_NULL_ARG;
184 *ctx_out = nullptr;
185
186 auto* ctx = static_cast<ufsecp_ctx*>(std::calloc(1, sizeof(ufsecp_ctx)));
187 if (!ctx) return UFSECP_ERR_INTERNAL;
188
189 ctx->last_err = UFSECP_OK;
190 ctx->last_msg[0] = '\0';
191
192 /* Run selftest once (cached globally by ensure_library_integrity) */
193 ctx->selftest_ok = secp256k1::fast::ensure_library_integrity(false);
194 if (!ctx->selftest_ok) {
195 std::free(ctx);
196 return UFSECP_ERR_SELFTEST;
197 }
198
199 *ctx_out = ctx;
200 return UFSECP_OK;
201}

321ufsecp_error_t ufsecp_pubkey_parse(ufsecp_ctx* ctx,
322 const uint8_t* input, size_t input_len,
323 uint8_t pubkey33_out[33]) {
324 if (!ctx || !input || !pubkey33_out) return UFSECP_ERR_NULL_ARG;
325 ctx_clear_err(ctx);
326
327 if (input_len == 33 && (input[0] == 0x02 || input[0] == 0x03)) {
328 auto p = point_from_compressed(input);
329 if (p.is_infinity())
330 return ctx_set_err(ctx, UFSECP_ERR_BAD_PUBKEY, "decompression failed");
331 point_to_compressed(p, pubkey33_out);
332 return UFSECP_OK;
333 }
334 if (input_len == 65 && input[0] == 0x04) {
335 std::array<uint8_t, 32> x_bytes, y_bytes;
336 std::memcpy(x_bytes.data(), input + 1, 32);
337 std::memcpy(y_bytes.data(), input + 33, 32);
338 auto x = FE::from_bytes(x_bytes);
339 auto y = FE::from_bytes(y_bytes);
340 auto p = Point::from_affine(x, y);
341 if (p.is_infinity())
342 return ctx_set_err(ctx, UFSECP_ERR_BAD_PUBKEY, "point at infinity");
343 point_to_compressed(p, pubkey33_out);
344 return UFSECP_OK;
345 }
346 return ctx_set_err(ctx, UFSECP_ERR_BAD_INPUT, "expected 33 or 65 byte pubkey");
347}

428ufsecp_error_t ufsecp_ecdsa_sig_from_der(ufsecp_ctx* ctx,
429 const uint8_t* der, size_t der_len,
430 uint8_t sig64_out[64]) {
431 if (!ctx || !der || !sig64_out) return UFSECP_ERR_NULL_ARG;
432 ctx_clear_err(ctx);
433
434 /* Minimal DER parser:
435 * 0x30 <total_len> 0x02 <r_len> <r_bytes...> 0x02 <s_len> <s_bytes...> */
436 if (der_len < 8 || der[0] != 0x30)
437 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: missing SEQUENCE");
438
439 size_t seq_len = der[1];
440 if (seq_len + 2 != der_len)
441 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: length mismatch");
442
443 size_t pos = 2;
444 /* Read R */
445 if (pos >= der_len || der[pos] != 0x02)
446 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: missing R INTEGER");
447 pos++;
448 if (pos >= der_len) return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: truncated");
449 size_t r_len = der[pos++];
450 if (r_len == 0 || pos + r_len > der_len)
451 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: R length");
452 const uint8_t* r_ptr = der + pos;
453 size_t r_data_len = r_len;
454 if (r_data_len > 0 && r_ptr[0] == 0x00) { r_ptr++; r_data_len--; }
455 pos += r_len;
456
457 /* Read S */
458 if (pos >= der_len || der[pos] != 0x02)
459 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: missing S INTEGER");
460 pos++;
461 if (pos >= der_len) return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: truncated");
462 size_t s_len = der[pos++];
463 if (s_len == 0 || pos + s_len > der_len)
464 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: S length");
465 const uint8_t* s_ptr = der + pos;
466 size_t s_data_len = s_len;
467 if (s_data_len > 0 && s_ptr[0] == 0x00) { s_ptr++; s_data_len--; }
468
469 if (r_data_len > 32 || s_data_len > 32)
470 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: component > 32 bytes");
471
472 std::memset(sig64_out, 0, 64);
473 std::memcpy(sig64_out + (32 - r_data_len), r_ptr, r_data_len);
474 std::memcpy(sig64_out + 32 + (32 - s_data_len), s_ptr, s_data_len);
475 return UFSECP_OK;
476}

751static void extkey_to_uf(const secp256k1::ExtendedKey& ek, ufsecp_bip32_key* out) {
752 auto serialized = ek.serialize();
753 std::memcpy(out->data, serialized.data(), 78);
754 out->is_private = ek.is_private ? 1 : 0;

758static secp256k1::ExtendedKey extkey_from_uf(const ufsecp_bip32_key* k) {
759 secp256k1::ExtendedKey ek{};
760 ek.depth = k->data[4];
761 std::memcpy(ek.parent_fingerprint.data(), k->data + 5, 4);
762 ek.child_number = (uint32_t(k->data[9]) << 24) | (uint32_t(k->data[10]) << 16) |
763 (uint32_t(k->data[11]) << 8) | uint32_t(k->data[12]);
764 std::memcpy(ek.chain_code.data(), k->data + 13, 32);
765 if (k->is_private) {
766 std::memcpy(ek.key.data(), k->data + 46, 32);
767 ek.is_private = true;
768 } else {
769 std::memcpy(ek.key.data(), k->data + 46, 32);
770 ek.is_private = false;
771 }
772 return ek;
773}
