103// Strict parser for secret keys: rejects 0, values >= n. No reduction.
104static inline bool scalar_parse_strict_nonzero(const uint8_t b[32], Scalar& out) {
105 std::array<uint8_t, 32> arr;
106 std::memcpy(arr.data(), b, 32);
107 return Scalar::parse_bytes_strict_nonzero(arr, out);

112 std::array<uint8_t, 32> arr;
113 std::memcpy(arr.data(), b, 32);
114 return Scalar::parse_bytes_strict(arr, out);
115}

122static inline Point point_from_compressed(const uint8_t pub[33]) {
123 // Strict: only accept 0x02/0x03 prefix, reject x >= p
124 if (pub[0] != 0x02 && pub[0] != 0x03) return Point::infinity();
125 FE x;
126 if (!FE::parse_bytes_strict(pub + 1, x)) return Point::infinity();
127
128 /* y^2 = x^3 + 7 */
129 auto x2 = x * x;
130 auto x3 = x2 * x;
131 auto y2 = x3 + FE::from_uint64(7);
132
133 /* sqrt via addition chain for (p+1)/4 */
134 auto t = y2;
135 auto a = t.square() * t;
136 auto b = a.square() * t;
137 auto c = b.square().square().square() * b;
138 auto d = c.square().square().square() * b;
139 auto e = d.square().square() * a;
140 auto f = e;
141 for (int i = 0; i < 11; ++i) f = f.square();
142 f = f * e;
143 auto g = f;
144 for (int i = 0; i < 22; ++i) g = g.square();
145 g = g * f;
146 auto h = g;
147 for (int i = 0; i < 44; ++i) h = h.square();
148 h = h * g;
149 auto j = h;
150 for (int i = 0; i < 88; ++i) j = j.square();
151 j = j * h;
152 auto k = j;
153 for (int i = 0; i < 44; ++i) k = k.square();
154 k = k * g;
155 auto m = k.square().square().square() * b;
156 auto y = m;
157 for (int i = 0; i < 23; ++i) y = y.square();
158 y = y * f;
159 for (int i = 0; i < 6; ++i) y = y.square();
160 y = y * a;
161 y = y.square().square();
162
163 // Verify sqrt: y^2 must equal y2 (reject if x has no valid y on curve)
164 if (y * y != y2) return Point::infinity();
165
166 auto y_bytes = y.to_bytes();
167 bool const y_is_odd = (y_bytes[31] & 1) != 0;
168 bool const want_odd = (pub[0] == 0x03);
169 if (y_is_odd != want_odd) {
170 y = FE::from_uint64(0) - y;
171}
172
173 return Point::from_affine(x, y);
174}

246ufsecp_error_t ufsecp_ctx_create(ufsecp_ctx** ctx_out) {
247 if (!ctx_out) return UFSECP_ERR_NULL_ARG;
248 *ctx_out = nullptr;
249
250 auto* ctx = static_cast<ufsecp_ctx*>(std::calloc(1, sizeof(ufsecp_ctx)));
251 if (!ctx) return UFSECP_ERR_INTERNAL;
252
253 ctx->last_err = UFSECP_OK;
254 ctx->last_msg[0] = '\0';
255
256 /* Run selftest once (cached globally by ensure_library_integrity) */
257 ctx->selftest_ok = secp256k1::fast::ensure_library_integrity(false);
258 if (!ctx->selftest_ok) {
259 std::free(ctx);
260 return UFSECP_ERR_SELFTEST;
261 }
262
263 *ctx_out = ctx;
264 return UFSECP_OK;
265}

315ufsecp_error_t ufsecp_seckey_negate(ufsecp_ctx* ctx, uint8_t privkey[32]) {
316 if (!ctx || !privkey) return UFSECP_ERR_NULL_ARG;
317 ctx_clear_err(ctx);
318 Scalar sk;

345 if (result.is_zero()) {

347 return ctx_set_err(ctx, UFSECP_ERR_ARITH, "tweak_add resulted in zero");
348 }
349 scalar_to_bytes(result, privkey);
350 secp256k1::detail::secure_erase(&result, sizeof(result));
351 return UFSECP_OK;

423ufsecp_error_t ufsecp_pubkey_parse(ufsecp_ctx* ctx,
424 const uint8_t* input, size_t input_len,
425 uint8_t pubkey33_out[33]) {
426 if (!ctx || !input || !pubkey33_out) return UFSECP_ERR_NULL_ARG;
427 ctx_clear_err(ctx);
428
429 if (input_len == 33 && (input[0] == 0x02 || input[0] == 0x03)) {
430 auto p = point_from_compressed(input);
431 if (p.is_infinity()) {
432 return ctx_set_err(ctx, UFSECP_ERR_BAD_PUBKEY, "decompression failed");
433}
434 point_to_compressed(p, pubkey33_out);
435 return UFSECP_OK;
436 }
437 if (input_len == 65 && input[0] == 0x04) {
438 std::array<uint8_t, 32> x_bytes, y_bytes;
439 std::memcpy(x_bytes.data(), input + 1, 32);
440 std::memcpy(y_bytes.data(), input + 33, 32);
441 // Strict: reject x >= p or y >= p
442 FE x, y;
443 if (!FE::parse_bytes_strict(x_bytes, x) ||
444 !FE::parse_bytes_strict(y_bytes, y)) {
445 return ctx_set_err(ctx, UFSECP_ERR_BAD_PUBKEY, "coordinate >= p");
446 }
447 // On-curve check: y^2 == x^3 + 7
448 auto lhs = y * y;
449 auto rhs = x * x * x + FE::from_uint64(7);
450 if (lhs != rhs) {
451 return ctx_set_err(ctx, UFSECP_ERR_BAD_PUBKEY, "point not on curve");
452 }
453 auto p = Point::from_affine(x, y);
454 if (p.is_infinity()) {
455 return ctx_set_err(ctx, UFSECP_ERR_BAD_PUBKEY, "point at infinity");
456}
457 point_to_compressed(p, pubkey33_out);
458 return UFSECP_OK;
459 }
460 return ctx_set_err(ctx, UFSECP_ERR_BAD_INPUT, "expected 33 or 65 byte pubkey");
461}

527ufsecp_error_t ufsecp_ecdsa_verify(ufsecp_ctx* ctx,
528 const uint8_t msg32[32],
529 const uint8_t sig64[64],
530 const uint8_t pubkey33[33]) {
531 if (!ctx || !msg32 || !sig64 || !pubkey33) return UFSECP_ERR_NULL_ARG;
532 ctx_clear_err(ctx);
533
534 std::array<uint8_t, 32> msg;
535 std::memcpy(msg.data(), msg32, 32);
536 std::array<uint8_t, 64> compact;
537 std::memcpy(compact.data(), sig64, 64);
538
539 secp256k1::ECDSASignature ecdsasig;
540 if (!secp256k1::ECDSASignature::parse_compact_strict(compact, ecdsasig)) {
541 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "non-canonical compact sig");
542 }
543 auto pk = point_from_compressed(pubkey33);
544 if (pk.is_infinity()) {
545 return ctx_set_err(ctx, UFSECP_ERR_BAD_PUBKEY, "invalid public key");
546 }
547
548 if (!secp256k1::ecdsa_verify(msg, pk, ecdsasig)) {
549 return ctx_set_err(ctx, UFSECP_ERR_VERIFY_FAIL, "ECDSA verify failed");
550 }
551
552 return UFSECP_OK;
553}

579ufsecp_error_t ufsecp_ecdsa_sig_from_der(ufsecp_ctx* ctx,
580 const uint8_t* der, size_t der_len,
581 uint8_t sig64_out[64]) {
582 if (!ctx || !der || !sig64_out) return UFSECP_ERR_NULL_ARG;
583 ctx_clear_err(ctx);
584
585 /* Strict DER parser for ECDSA secp256k1 signatures.
586 * Format: 0x30 <total_len> 0x02 <r_len> <r_bytes...> 0x02 <s_len> <s_bytes...>
587 *
588 * Enforces:
589 * - Single-byte length encoding only (no long form)
590 * - No negative integers (high bit of first data byte must be 0)
591 * - No unnecessary leading zero padding
592 * - Exact total length (no trailing bytes)
593 * - r, s must be in [1, n-1] (canonical, nonzero)
594 * - Max total DER length: 72 bytes */
595
596 /* Max DER ECDSA sig: 2 + 2 + 33 + 2 + 33 = 72 */
597 if (der_len < 8 || der_len > 72 || der[0] != 0x30) {
598 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: missing/oversized SEQUENCE");
599 }
600
601 /* Reject long-form length encoding (bit 7 set = multi-byte length) */
602 if (der[1] & 0x80) {
603 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: long-form length");
604 }
605
606 size_t const seq_len = der[1];
607 if (seq_len + 2 != der_len) {
608 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: length mismatch");
609 }
610
611 size_t pos = 2;
612
613 /* --- Helper lambda: parse one INTEGER component strictly --- */
614 auto parse_int = [&](const char* name, const uint8_t*& out_ptr,
615 size_t& out_len) -> ufsecp_error_t {
616 if (pos >= der_len || der[pos] != 0x02) {
617 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: missing INTEGER");
618 }
619 pos++;
620 if (pos >= der_len) {
621 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: truncated");
622 }
623 /* Reject long-form length for component */
624 if (der[pos] & 0x80) {
625 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: long-form int length");
626 }
627 size_t const int_len = der[pos++];
628 if (int_len == 0 || pos + int_len > der_len) {
629 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: int length out of bounds");
630 }
631 /* Reject negative: high bit set on first data byte means negative in DER */
632 if (der[pos] & 0x80) {
633 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: negative integer");
634 }
635 /* Reject unnecessary leading zero: 0x00 prefix only valid when next byte
636 * has high bit set (positive number needs padding to stay positive).
637 * If next byte has high bit clear, the 0x00 is superfluous padding. */
638 if (int_len > 1 && der[pos] == 0x00 && !(der[pos + 1] & 0x80)) {
639 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: unnecessary leading zero");
640 }
641
642 out_ptr = der + pos;
643 out_len = int_len;
644 /* Strip valid leading zero pad (high bit of next byte is set) */
645 if (out_len > 0 && out_ptr[0] == 0x00) { out_ptr++; out_len--; }
646 pos += int_len;
647 (void)name;
648 return UFSECP_OK;
649 };
650
651 /* Read R */
652 const uint8_t* r_ptr = nullptr;
653 size_t r_data_len = 0;
654 {
655 auto rc = parse_int("R", r_ptr, r_data_len);
656 if (rc != UFSECP_OK) return rc;
657 }
658
659 /* Read S */
660 const uint8_t* s_ptr = nullptr;
661 size_t s_data_len = 0;
662 {
663 auto rc = parse_int("S", s_ptr, s_data_len);
664 if (rc != UFSECP_OK) return rc;
665 }
666
667 /* Reject trailing bytes after S (must consume entire SEQUENCE) */
668 if (pos != der_len) {
669 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: trailing bytes");
670 }
671
672 if (r_data_len > 32 || s_data_len > 32) {
673 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: component > 32 bytes");
674 }
675
676 /* Build compact sig64 (big-endian, right-aligned in 32-byte slots) */
677 std::memset(sig64_out, 0, 64);
678 /* Explicit null checks for static analyzer (r_ptr/s_ptr guaranteed non-null
679 * when *_data_len > 0 by parse_int() success, but SonarCloud can't track it) */
680 if (r_data_len > 0 && r_ptr) {
681 std::memcpy(sig64_out + (32 - r_data_len), r_ptr, r_data_len);
682 }
683 if (s_data_len > 0 && s_ptr) {
684 std::memcpy(sig64_out + 32 + (32 - s_data_len), s_ptr, s_data_len);
685 }
686
687 /* Range check: r and s must be in [1, n-1] (strict nonzero, no reduce) */
688 Scalar r_sc, s_sc;
689 if (!Scalar::parse_bytes_strict_nonzero(sig64_out, r_sc) ||
690 !Scalar::parse_bytes_strict_nonzero(sig64_out + 32, s_sc)) {
691 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "bad DER: r or s out of range [1,n-1]");
692 }
693
694 return UFSECP_OK;
695}

699ufsecp_error_t ufsecp_ecdsa_sign_recoverable(ufsecp_ctx* ctx,
700 const uint8_t msg32[32],
701 const uint8_t privkey[32],
702 uint8_t sig64_out[64],
703 int* recid_out) {
704 if (!ctx || !msg32 || !privkey || !sig64_out || !recid_out) {
705 return UFSECP_ERR_NULL_ARG;
706}

708
709 std::array<uint8_t, 32> msg;
710 std::memcpy(msg.data(), msg32, 32);
711 Scalar sk;
712 if (!scalar_parse_strict_nonzero(privkey, sk)) {
713 return ctx_set_err(ctx, UFSECP_ERR_BAD_KEY, "privkey is zero or >= n");
714 }
715
716 // NOTE: No ct::ecdsa_sign_recoverable exists yet. Using fast path with
717 // zeroization. Recovery signing is inherently non-CT due to recid computation.
718 // TODO: Implement ct::ecdsa_sign_recoverable when CT recovery is needed.
719 auto rsig = secp256k1::ecdsa_sign_recoverable(msg, sk);
720 secp256k1::detail::secure_erase(&sk, sizeof(sk));
721 auto normalized = rsig.sig.normalize();
722 auto compact = normalized.to_compact();
723 std::memcpy(sig64_out, compact.data(), 64);
724 *recid_out = rsig.recid;
725 return UFSECP_OK;
726}

728ufsecp_error_t ufsecp_ecdsa_recover(ufsecp_ctx* ctx,
729 const uint8_t msg32[32],
730 const uint8_t sig64[64],
731 int recid,
732 uint8_t pubkey33_out[33]) {
733 if (!ctx || !msg32 || !sig64 || !pubkey33_out) return UFSECP_ERR_NULL_ARG;
734 ctx_clear_err(ctx);
735
736 if (recid < 0 || recid > 3) {
737 return ctx_set_err(ctx, UFSECP_ERR_BAD_INPUT, "recid must be 0..3");
738 }
739
740 std::array<uint8_t, 32> msg;
741 std::memcpy(msg.data(), msg32, 32);
742 std::array<uint8_t, 64> compact;
743 std::memcpy(compact.data(), sig64, 64);
744
745 secp256k1::ECDSASignature ecdsasig;
746 if (!secp256k1::ECDSASignature::parse_compact_strict(compact, ecdsasig)) {
747 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "non-canonical compact sig");
748 }
749
750 auto [point, ok] = secp256k1::ecdsa_recover(msg, ecdsasig, recid);
751 if (!ok) {
752 return ctx_set_err(ctx, UFSECP_ERR_VERIFY_FAIL, "recovery failed");
753 }
754
755 point_to_compressed(point, pubkey33_out);
756 return UFSECP_OK;
757}

819ufsecp_error_t ufsecp_schnorr_verify(ufsecp_ctx* ctx,
820 const uint8_t msg32[32],
821 const uint8_t sig64[64],
822 const uint8_t pubkey_x[32]) {
823 if (!ctx || !msg32 || !sig64 || !pubkey_x) return UFSECP_ERR_NULL_ARG;
824 ctx_clear_err(ctx);
825
826 // BIP-340 strict parse: reject non-canonical r >= p, s >= n, or s == 0
827 secp256k1::SchnorrSignature schnorr_sig;
828 if (!secp256k1::SchnorrSignature::parse_strict(sig64, schnorr_sig)) {
829 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "Non-canonical Schnorr sig (r>=p or s>=n)");
830 }
831
832 // BIP-340 strict: reject pubkey x >= p
833 FE pk_fe;
834 if (!FE::parse_bytes_strict(pubkey_x, pk_fe)) {
835 return ctx_set_err(ctx, UFSECP_ERR_BAD_PUBKEY, "Non-canonical pubkey (x>=p)");
836 }
837
838 std::array<uint8_t, 32> pk_arr, msg_arr;
839 std::memcpy(pk_arr.data(), pubkey_x, 32);
840 std::memcpy(msg_arr.data(), msg32, 32);
841
842 if (!secp256k1::schnorr_verify(pk_arr, msg_arr, schnorr_sig)) {
843 return ctx_set_err(ctx, UFSECP_ERR_VERIFY_FAIL, "Schnorr verify failed");
844}
845
846 return UFSECP_OK;
847}

1456 std::memcpy(batch[i].message.data(), e + 32, 32);
1457 if (!secp256k1::SchnorrSignature::parse_strict(e + 64, batch[i].signature)) {
1458 return ctx_set_err(ctx, UFSECP_ERR_BAD_SIG, "invalid Schnorr sig in batch");
1459 }
1460 }
1461 if (!secp256k1::schnorr_batch_verify(batch)) {
1462 return ctx_set_err(ctx, UFSECP_ERR_VERIFY_FAIL, "batch verify failed");
1463 }

1701ufsecp_error_t ufsecp_musig2_start_sign_session(
1702 ufsecp_ctx* ctx,
1703 const uint8_t aggnonce[UFSECP_MUSIG2_AGGNONCE_LEN],
1704 const uint8_t keyagg[UFSECP_MUSIG2_KEYAGG_LEN],
1705 const uint8_t msg32[32],
1706 uint8_t session_out[UFSECP_MUSIG2_SESSION_LEN]) {
1707 if (!ctx || !aggnonce || !keyagg || !msg32 || !session_out) return UFSECP_ERR_NULL_ARG;
1708 ctx_clear_err(ctx);
1709 /* Deserialize agg nonce */
1710 secp256k1::MuSig2AggNonce an;
1711 an.R1 = point_from_compressed(aggnonce);
1712 if (an.R1.is_infinity()) {
1713 return ctx_set_err(ctx, UFSECP_ERR_BAD_INPUT, "invalid agg nonce R1");
1714 }
1715 an.R2 = point_from_compressed(aggnonce + 33);
1716 if (an.R2.is_infinity()) {
1717 return ctx_set_err(ctx, UFSECP_ERR_BAD_INPUT, "invalid agg nonce R2");
1718 }
1719 /* Deserialize key agg context */
1720 secp256k1::MuSig2KeyAggCtx kagg;
1721 uint32_t nk = 0;
1722 std::memcpy(&nk, keyagg, 4);
1723 kagg.Q_negated = (keyagg[4] != 0);
1724 kagg.Q = point_from_compressed(keyagg + 5);
1725 if (kagg.Q.is_infinity()) {
1726 return ctx_set_err(ctx, UFSECP_ERR_BAD_KEY, "invalid aggregated key");
1727 }
1728 auto qc = kagg.Q.to_compressed();
1729 std::memcpy(kagg.Q_x.data(), qc.data() + 1, 32);
1730 for (uint32_t i = 0; i < nk && (38u + static_cast<size_t>(i+1)*32u <= UFSECP_MUSIG2_KEYAGG_LEN); ++i) {
1731 Scalar s;
1732 if (!scalar_parse_strict(keyagg + 38 + static_cast<size_t>(i) * 32, s)) {
1733 return ctx_set_err(ctx, UFSECP_ERR_BAD_INPUT, "invalid key coefficient in keyagg");
1734 }
1735 kagg.key_coefficients.push_back(s);
1736 }
1737 std::array<uint8_t, 32> msg_arr;
1738 std::memcpy(msg_arr.data(), msg32, 32);
1739 auto sess = secp256k1::musig2_start_sign_session(an, kagg, msg_arr);
1740 /* Serialize session: R(33) | b(32) | e(32) | R_negated(1) = 98 bytes */
1741 std::memset(session_out, 0, UFSECP_MUSIG2_SESSION_LEN);
1742 point_to_compressed(sess.R, session_out);
1743 scalar_to_bytes(sess.b, session_out + 33);
1744 scalar_to_bytes(sess.e, session_out + 65);
1745 session_out[97] = sess.R_negated ? 1 : 0;
1746 return UFSECP_OK;
1747}

1902 ufsecp_ctx* ctx,
1903 uint32_t participant_id, uint32_t threshold, uint32_t num_participants,
1904 const uint8_t seed[32],

1908 return UFSECP_ERR_NULL_ARG;
1909 }
1910 ctx_clear_err(ctx);

1912 return ctx_set_err(ctx, UFSECP_ERR_BAD_INPUT, "invalid threshold");
1913 }
1914 std::array<uint8_t, 32> seed_arr;
1915 std::memcpy(seed_arr.data(), seed, 32);

1917 participant_id, threshold, num_participants, seed_arr);
1918 secp256k1::detail::secure_erase(seed_arr.data(), 32);

1920 const size_t coeff_count = commit.coeffs.size();
1921 const size_t needed_commits = 8 + coeff_count * 33;

1929 point_to_compressed(commit.coeffs[i], commits_out + 8 + i * 33);
1930
1931 }

1934 const size_t needed_shares = shares.size() * UFSECP_FROST_SHARE_LEN;
1935 if (*shares_len < needed_shares) {
1936 return ctx_set_err(ctx, UFSECP_ERR_BUF_TOO_SMALL, "shares buffer too small");
1937 }

1939 uint8_t* s = shares_out + i * UFSECP_FROST_SHARE_LEN;
1940 std::memcpy(s, &shares[i].from, 4);
1941 scalar_to_bytes(shares[i].value, s + 4);
1942 }

1951ufsecp_error_t ufsecp_frost_keygen_finalize(
1952 ufsecp_ctx* ctx,
1953 uint32_t participant_id,
1954 const uint8_t* all_commits, size_t commits_len,
1955 const uint8_t* received_shares, size_t shares_len,
1956 uint32_t threshold, uint32_t num_participants,
1957 uint8_t keypkg_out[UFSECP_FROST_KEYPKG_LEN]) {
1958 if (!ctx || !all_commits || !received_shares || !keypkg_out) return UFSECP_ERR_NULL_ARG;
1959 ctx_clear_err(ctx);
1960 /* Deserialize commitments */
1961 std::vector<secp256k1::FrostCommitment> commits;
1962 size_t pos = 0;
1963 while (pos < commits_len) {
1964 secp256k1::FrostCommitment fc;
1965 uint32_t cc = 0;
1966 if (pos + 8 > commits_len) {
1967 return ctx_set_err(ctx, UFSECP_ERR_BAD_INPUT, "truncated commit header");
1968 }
1969 std::memcpy(&cc, all_commits + pos, 4); pos += 4;
1970 std::memcpy(&fc.from, all_commits + pos, 4); pos += 4;
1971 if (pos + static_cast<size_t>(cc) * 33 > commits_len) {
1972 return ctx_set_err(ctx, UFSECP_ERR_BAD_INPUT, "truncated commit coefficients");
1973 }
1974 for (uint32_t j = 0; j < cc; ++j) {
1975 auto pt = point_from_compressed(all_commits + pos);
1976 if (pt.is_infinity()) {
1977 return ctx_set_err(ctx, UFSECP_ERR_BAD_INPUT, "invalid commitment coefficient");
1978 }
1979 fc.coeffs.push_back(pt);
1980 pos += 33;
1981 }
1982 commits.push_back(std::move(fc));
1983 }
1984 /* Deserialize shares */
1985 const size_t n_shares = shares_len / UFSECP_FROST_SHARE_LEN;
1986 std::vector<secp256k1::FrostShare> shares(n_shares);
1987 for (size_t i = 0; i < n_shares; ++i) {
1988 const uint8_t* s = received_shares + i * UFSECP_FROST_SHARE_LEN;
1989 std::memcpy(&shares[i].from, s, 4);
1990 shares[i].id = participant_id;
1991 Scalar v;
1992 if (!scalar_parse_strict(s + 4, v)) {
1993 return ctx_set_err(ctx, UFSECP_ERR_BAD_INPUT, "invalid share scalar");
1994 }
1995 shares[i].value = v;
1996 }
1997 auto [kp, ok] = secp256k1::frost_keygen_finalize(
1998 participant_id, commits, shares, threshold, num_participants);
1999 if (!ok) {
2000 return ctx_set_err(ctx, UFSECP_ERR_INTERNAL, "FROST keygen finalize failed");
2001 }
2002 // Erase secret shares
2003 for (auto& s : shares) {
2004 secp256k1::detail::secure_erase(&s.value, sizeof(s.value));
2005 }
2006 /* Serialize FrostKeyPackage: id(4) | threshold(4) | num_participants(4) |
2007 signing_share(32) | verification_share(33) | group_public_key(33) = 110 bytes */
2008 std::memset(keypkg_out, 0, UFSECP_FROST_KEYPKG_LEN);
2009 std::memcpy(keypkg_out, &kp.id, 4);
2010 std::memcpy(keypkg_out + 4, &kp.threshold, 4);
2011 std::memcpy(keypkg_out + 8, &kp.num_participants, 4);
2012 scalar_to_bytes(kp.signing_share, keypkg_out + 12);
2013 point_to_compressed(kp.verification_share, keypkg_out + 44);
2014 point_to_compressed(kp.group_public_key, keypkg_out + 77);
2015 secp256k1::detail::secure_erase(&kp.signing_share, sizeof(kp.signing_share));
2016 return UFSECP_OK;
2017}

2724ufsecp_error_t ufsecp_zk_range_prove(
2725 ufsecp_ctx* ctx,
2726 uint64_t value,
2727 const uint8_t blinding[32],
2728 const uint8_t commitment33[33],
2729 const uint8_t aux_rand[32],
2730 uint8_t* proof_out, size_t* proof_len) {
2731 if (!ctx || !blinding || !commitment33 || !aux_rand || !proof_out || !proof_len) {
2732 return UFSECP_ERR_NULL_ARG;
2733 }
2734 ctx_clear_err(ctx);
2735 Scalar b;
2736 if (!scalar_parse_strict(blinding, b)) {
2737 return ctx_set_err(ctx, UFSECP_ERR_BAD_INPUT, "blinding >= n");
2738 }
2739 const SecureEraseGuard<Scalar> blinding_guard(&b);
2740 auto commit_pt = point_from_compressed(commitment33);

2742 return ctx_set_err(ctx, UFSECP_ERR_BAD_INPUT, "invalid commitment point");

2748 /* Serialize range proof: A(33)+S(33)+T1(33)+T2(33)+tau_x(32)+mu(32)+t_hat(32)+L[6]*33+R[6]*33+a(32)+b(32) */

2755 auto c = p.to_compressed();
2756 std::memcpy(proof_out + off, c.data(), 33);
2757 off += 33;
2758 };
2759 auto write_scalar = [&](const Scalar& s) {
2760 scalar_to_bytes(s, proof_out + off);
2761 off += 32;
2762 };
2763 write_point(rp.A); write_point(rp.S);
2764 write_point(rp.T1); write_point(rp.T2);
2765 write_scalar(rp.tau_x); write_scalar(rp.mu); write_scalar(rp.t_hat);
2766 for (int i = 0; i < 6; ++i) write_point(rp.L[i]);
2767 for (int i = 0; i < 6; ++i) write_point(rp.R[i]);
2768 write_scalar(rp.a); write_scalar(rp.b);

2771}

2773ufsecp_error_t ufsecp_zk_range_verify(
2774 ufsecp_ctx* ctx,
2775 const uint8_t commitment33[33],
2776 const uint8_t* proof, size_t proof_len) {
2777 if (!ctx || !commitment33 || !proof) return UFSECP_ERR_NULL_ARG;
2778 ctx_clear_err(ctx);
2779 /* Deserialize range proof */
2780 const size_t expected = 33*4 + 32*3 + 6*33 + 6*33 + 32*2;
2781 if (proof_len < expected) {
2782 return ctx_set_err(ctx, UFSECP_ERR_BAD_INPUT, "range proof too short");
2783 }
2784 secp256k1::zk::RangeProof rp;
2785 size_t off = 0;
2786 bool point_ok = true;
2787 auto read_point = [&]() -> Point {
2788 auto p = point_from_compressed(proof + off);
2789 if (p.is_infinity()) point_ok = false;
2790 off += 33;
2791 return p;
2792 };
2793 bool scalar_ok = true;
2794 auto read_scalar = [&]() -> Scalar {
2795 Scalar s;
2796 if (!scalar_parse_strict(proof + off, s)) {
2797 scalar_ok = false;
2798 }
2799 off += 32;
2800 return s;
2801 };
2802 rp.A = read_point(); rp.S = read_point();
2803 rp.T1 = read_point(); rp.T2 = read_point();
2804 rp.tau_x = read_scalar(); rp.mu = read_scalar(); rp.t_hat = read_scalar();
2805 for (int i = 0; i < 6; ++i) rp.L[i] = read_point();
2806 for (int i = 0; i < 6; ++i) rp.R[i] = read_point();
2807 rp.a = read_scalar(); rp.b = read_scalar();
2808 if (!point_ok) {
2809 return ctx_set_err(ctx, UFSECP_ERR_BAD_INPUT, "invalid point in range proof");
2810 }
2811 if (!scalar_ok) {
2812 return ctx_set_err(ctx, UFSECP_ERR_BAD_INPUT, "invalid scalar in range proof");
2813 }
2814 auto commit_pt = point_from_compressed(commitment33);
2815 if (commit_pt.is_infinity()) {
2816 return ctx_set_err(ctx, UFSECP_ERR_BAD_INPUT, "invalid commitment point");
2817 }
2818 auto commit = secp256k1::PedersenCommitment{commit_pt};
2819 if (!secp256k1::zk::range_verify(commit, rp)) {
2820 return ctx_set_err(ctx, UFSECP_ERR_VERIFY_FAIL, "range proof failed");
2821 }
2822 return UFSECP_OK;
2823}
