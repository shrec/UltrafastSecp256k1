4// ============================================================================
5// AVX2 / AVX-512 SIMD Field Arithmetic — secp256k1
6// ============================================================================
7// Batch field operations using x86 SIMD intrinsics for 4x (AVX2) or
8// 8x (AVX-512) parallel field element processing.
9//
10// Architecture:
11// - Runtime CPUID detection: avx2_available(), avx512_available()
12// - Batch API processes N field elements in parallel
13// - Falls back to scalar when SIMD not available
14//
15// Performance model:
16// - AVX2 (256-bit): 4 field ops in parallel → ~3x throughput for batch work
17// - AVX-512 (512-bit): 8 field ops → ~5-6x throughput
18// - Only beneficial for batch operations (batch verify, multi-scalar mul)
19// - Single-element ops are faster with scalar code (pipeline fill overhead)
20//
21// Usage:
22// if (secp256k1::simd::avx2_available()) {
23// secp256k1::simd::batch_field_add_avx2(out, a, b, count);
24// }
25// ============================================================================

50// Check if AVX2 is available at runtime
51inline bool avx2_available() noexcept {
52#ifdef SECP256K1_X86_TARGET
53 #ifdef _MSC_VER
54 int info[4];
55 __cpuidex(info, 7, 0);
56 return (info[1] & (1 << 5)) != 0; // EBX bit 5 = AVX2
57 #elif defined(__GNUC__) || defined(__clang__)
58 unsigned int eax, ebx, ecx, edx;
59 if (__get_cpuid_count(7, 0, &eax, &ebx, &ecx, &edx)) {
60 return (ebx & (1 << 5)) != 0;
61 }
62 return false;
63 #endif
64#else
65 return false;
66#endif
67}

90enum class SimdTier : int {
91 SCALAR = 0, // No SIMD, scalar fallback
92 AVX2 = 1, // AVX2 (256-bit, 4-way)
93 AVX512 = 2, // AVX-512 (512-bit, 8-way)

96// Detect best available SIMD tier
97inline SimdTier detect_simd_tier() noexcept {
98 if (avx512_available()) return SimdTier::AVX512;
99 if (avx2_available()) return SimdTier::AVX2;
100 return SimdTier::SCALAR;
101}
