195// Optimized precomputed scalar - ZERO runtime calculations + RLE compression!
196// Uses Run-Length Encoding to skip consecutive zeros (doubles without adds)
197struct PrecomputedScalarOptimized {
198 // Each step: N doubles followed by 0-2 additions
199 struct Step {
200 uint16_t num_doubles; // How many consecutive doubles before additions
201
202 // Addition info for k₁ (Q table)
203 uint8_t idx1; // Point index [0..table_size-1], 0xFF = skip
204 bool neg1; // Negate Y coordinate?
205
206 // Addition info for k₂ (ψ(Q) table)
207 uint8_t idx2; // Point index [0..table_size-1], 0xFF = skip
208 bool neg2; // Negate Y coordinate?
209
210 Step() : num_doubles(0), idx1(0xFF), neg1(false), idx2(0xFF), neg2(false) {}
211 };
212
213 Scalar k1, k2;
214 bool neg1, neg2;
215 std::vector<Step> steps; // ~100-130 steps (instead of 256 iterations!)
216 unsigned window_bits;
217
218 bool is_valid() const { return window_bits >= 4 && window_bits <= 7 && !steps.empty(); }
219};
