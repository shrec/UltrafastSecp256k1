5// ============================================================================
6// Coin Parameters — constexpr definitions for all secp256k1-based coins
7// ============================================================================
8// Each CoinParams holds:
9// - Network prefixes (P2PKH version byte, WIF prefix)
10// - Bech32 HRP (human-readable part)
11// - BIP-44 coin_type
12// - Address hash algorithm (HASH160 vs Keccak-256)
13// - Feature flags (supports_segwit, supports_taproot, etc.)
14// - Display name + ticker
15//
16// All data is constexpr — zero runtime cost, no heap allocation.
17// ============================================================================
18
19#include <cstdint>
20#include <array>
21
22namespace secp256k1::coins {
23
24// ── Address Hash Algorithm ───────────────────────────────────────────────────
25
26enum class AddressHash : std::uint8_t {
27 HASH160, // RIPEMD160(SHA256(pubkey)) — Bitcoin, Litecoin, etc.
28 KECCAK256, // Keccak-256(pubkey[1:]) — Ethereum, BNB, etc.
29 BLAKE2B_160, // BLAKE2b-160 — reserved for future use
30};
31
32// ── Address Encoding ─────────────────────────────────────────────────────────
33
34enum class AddressEncoding : std::uint8_t {
35 BASE58CHECK, // Base58Check (P2PKH, P2SH)
36 BECH32, // Bech32 / Bech32m (SegWit)
37 EIP55, // Ethereum EIP-55 mixed-case checksum
38 CASHADDR, // CashAddr (Bitcoin Cash)
39};
40
41// ── Feature Flags ────────────────────────────────────────────────────────────
42
43struct CoinFeatures {
44 bool supports_segwit : 1; // SegWit (P2WPKH, Bech32)
45 bool supports_taproot : 1; // Taproot (P2TR, Bech32m)
46 bool supports_p2sh : 1; // P2SH (pay-to-script-hash)
47 bool compressed_only : 1; // Only compressed pubkeys allowed
48 bool uses_evm : 1; // EVM-compatible (Ethereum-like)
49 bool uses_schnorr : 1; // Native Schnorr support (BIP-340)
50};
51
52// ── Coin Parameters ──────────────────────────────────────────────────────────
53
54struct CoinParams {
55 // Display
56 const char* name; // "Bitcoin", "Litecoin", etc.
57 const char* ticker; // "BTC", "LTC", etc.
58
59 // Network prefixes
60 std::uint8_t p2pkh_version; // P2PKH address version byte
61 std::uint8_t p2pkh_version_test; // P2PKH testnet version byte
62 std::uint8_t p2sh_version; // P2SH version byte
63 std::uint8_t wif_prefix; // WIF private key prefix
64 std::uint8_t wif_prefix_test; // WIF testnet prefix
65
66 // Bech32
67 const char* bech32_hrp; // "bc", "ltc", "tb" or nullptr
68 const char* bech32_hrp_test; // Testnet HRP or nullptr
69
70 // BIP-44
71 std::uint32_t coin_type; // BIP-44 coin_type index
72
73 // Addressing
74 AddressHash hash_algo; // Which hash for address derivation
75 AddressEncoding default_encoding; // Primary address format
76
77 // Extended key versions (BIP-32)
78 std::uint32_t xprv_version; // xprv serialization magic (4 bytes)
79 std::uint32_t xpub_version; // xpub serialization magic (4 bytes)
80
81 // Features
82 CoinFeatures features;
83};

627// Find coin by ticker string (case-sensitive, returns nullptr if not found)
628inline const CoinParams* find_by_ticker(const char* ticker) {
629 for (std::size_t i = 0; i < ALL_COINS_COUNT; ++i) {
630 const char* a = ALL_COINS[i]->ticker;
631 const char* b = ticker;
632 bool match = true;
633 while (*a && *b) {
634 if (*a != *b) { match = false; break; }
635 ++a; ++b;
636 }
637 if (match && *a == *b) return ALL_COINS[i];
638 }
639 return nullptr;
640}
