25// ─── Declassify / Classify Markers ───────────────────────────────────────────
26// For constant-time verification with Valgrind (memcheck) or MSAN.
27//
28// SECP256K1_CLASSIFY(ptr, len) — Mark memory as secret (undefined).
29// Call on inputs before CT operations.
30// SECP256K1_DECLASSIFY(ptr, len) — Mark memory as public (defined).
31// Call on outputs after CT operations.
32//
33// Under normal compilation these are no-ops. When compiled with:
34// -DSECP256K1_CT_VALGRIND=1 and Valgrind headers available:
35// maps to VALGRIND_MAKE_MEM_{UNDEFINED,DEFINED}
36//
37// Usage in tests:
38// Scalar k;
39// SECP256K1_CLASSIFY(&k, sizeof(k)); // treat k as secret
40// Point R = ct::scalar_mul(G, k); // CT operation under test
41// SECP256K1_DECLASSIFY(&R, sizeof(R)); // declassify result for comparison
42//
43// If valgrind reports a "conditional jump depends on uninitialised value"
44// error BETWEEN classify and declassify, the code has a CT violation.
45// ─────────────────────────────────────────────────────────────────────────────
46
47#if defined(SECP256K1_CT_VALGRIND) && SECP256K1_CT_VALGRIND
48 #include <valgrind/memcheck.h>
49 #define SECP256K1_CLASSIFY(ptr, len) VALGRIND_MAKE_MEM_UNDEFINED((ptr), (len))
50 #define SECP256K1_DECLASSIFY(ptr, len) VALGRIND_MAKE_MEM_DEFINED((ptr), (len))
51#else
52 #define SECP256K1_CLASSIFY(ptr, len) ((void)(ptr), (void)(len))
53 #define SECP256K1_DECLASSIFY(ptr, len) ((void)(ptr), (void)(len))
54#endif

58// ─── Compiler barrier ────────────────────────────────────────────────────────
59// Prevents compiler from optimizing away branchless patterns.
60// Uses inline asm (GCC/Clang) or volatile (MSVC) to create optimization barrier.
61
62#if defined(__GNUC__) || defined(__clang__)
63 // Force value through a register so compiler cannot reason about it
64 inline void value_barrier(std::uint64_t& v) noexcept {
65 asm volatile("" : "+r"(v) : : "memory");
66 }
67 inline void value_barrier(std::uint32_t& v) noexcept {
68 asm volatile("" : "+r"(v) : : "memory");
69 }
70#else
71 // MSVC: volatile prevents optimization of subsequent operations
72 inline void value_barrier(std::uint64_t& v) noexcept {
73 volatile std::uint64_t sink = v;
74 v = sink;
75 }
76 inline void value_barrier(std::uint32_t& v) noexcept {
77 volatile std::uint32_t sink = v;
78 v = sink;
79 }
80#endif

84// Returns 0xFFFFFFFFFFFFFFFF if v == 0, else 0x0000000000000000
85inline std::uint64_t is_zero_mask(std::uint64_t v) noexcept {
86 // ~(v | -v) has MSB set iff v == 0
87 std::uint64_t nv = -v;
88 value_barrier(nv);
89 return static_cast<std::uint64_t>(
90 -static_cast<std::int64_t>((~(v | nv)) >> 63));
91}
92
93// Returns 0xFFFFFFFFFFFFFFFF if v != 0, else 0x0000000000000000
94inline std::uint64_t is_nonzero_mask(std::uint64_t v) noexcept {
95 return ~is_zero_mask(v);
96}
97
98// Returns 0xFFFFFFFFFFFFFFFF if a == b, else 0x0000000000000000
99inline std::uint64_t eq_mask(std::uint64_t a, std::uint64_t b) noexcept {
100 return is_zero_mask(a ^ b);
101}

103// Returns 0xFFFFFFFFFFFFFFFF if flag is true (nonzero), else 0
104inline std::uint64_t bool_to_mask(bool flag) noexcept {
105 std::uint64_t v = static_cast<std::uint64_t>(flag);
106 value_barrier(v);
107 return -v;
108}

110// Returns 0xFFFFFFFFFFFFFFFF if a < b (unsigned), else 0
111// Uses the borrow bit from subtraction
112inline std::uint64_t lt_mask(std::uint64_t a, std::uint64_t b) noexcept {
113 // If a < b, then (a - b) borrows, so bit 64 of the extended result is 1
114 // We compute this as: a < b <=> ~a >= b <=> we check borrow
115 std::uint64_t diff = a - b;
116 // Borrow occurred iff a < b. The borrow is in the "carry out" position.
117 // For unsigned: a < b iff the subtract wraps, iff (a ^ ((a ^ b) | (diff ^ a))) has MSB set
118 std::uint64_t borrow = (a ^ ((a ^ b) | (diff ^ a))) >> 63;
119 value_barrier(borrow);
120 return -borrow;
121}

123// ─── Conditional move (CT) ───────────────────────────────────────────────────
124// if (flag) *dst = *src; — constant time, no branch
125
126inline void cmov64(std::uint64_t* dst, const std::uint64_t* src,
127 std::uint64_t mask) noexcept {
128 // dst = (src & mask) | (dst & ~mask)
129 *dst ^= (*dst ^ *src) & mask;
130}

132// Conditional move for 4-limb (256-bit) values
133inline void cmov256(std::uint64_t dst[4], const std::uint64_t src[4],
134 std::uint64_t mask) noexcept {
135 dst[0] ^= (dst[0] ^ src[0]) & mask;
136 dst[1] ^= (dst[1] ^ src[1]) & mask;
137 dst[2] ^= (dst[2] ^ src[2]) & mask;
138 dst[3] ^= (dst[3] ^ src[3]) & mask;
139}

141// ─── Conditional swap (CT) ───────────────────────────────────────────────────
142// if (flag) swap(*a, *b); — constant time
143
144inline void cswap256(std::uint64_t a[4], std::uint64_t b[4],
145 std::uint64_t mask) noexcept {
146 for (int i = 0; i < 4; ++i) {
147 std::uint64_t diff = (a[i] ^ b[i]) & mask;
148 a[i] ^= diff;
149 b[i] ^= diff;
150 }
151}

153// ─── Constant-time select ────────────────────────────────────────────────────
154// Returns a if mask == 0xFFF...F, else b (mask must be all-zeros or all-ones)
155
156inline std::uint64_t ct_select(std::uint64_t a, std::uint64_t b,
157 std::uint64_t mask) noexcept {
158 return (a & mask) | (b & ~mask);
159}

161// ─── Constant-time table lookup ──────────────────────────────────────────────
162// Always reads ALL entries. No secret-dependent memory access pattern.
163// table: pointer to N entries of 'stride' bytes each
164// index: which entry to select (0-based)
165// out: output buffer (stride bytes)
166
167inline void ct_lookup(const void* table, std::size_t count,
168 std::size_t stride, std::size_t index,
169 void* out) noexcept {
170 // Zero the output
171 auto* dst = static_cast<std::uint8_t*>(out);
172 for (std::size_t b = 0; b < stride; ++b) dst[b] = 0;
173
174 const auto* src = static_cast<const std::uint8_t*>(table);
175 for (std::size_t i = 0; i < count; ++i) {
176 std::uint64_t mask = eq_mask(static_cast<std::uint64_t>(i),
177 static_cast<std::uint64_t>(index));
178 auto mask8 = static_cast<std::uint8_t>(mask & 0xFF);
179 for (std::size_t b = 0; b < stride; ++b) {
180 dst[b] |= src[i * stride + b] & mask8;
181 }
182 }
183}

185// Specialized: CT lookup for 4-limb (256-bit) entries
186inline void ct_lookup_256(const std::uint64_t table[][4], std::size_t count,
187 std::size_t index, std::uint64_t out[4]) noexcept {
188 out[0] = out[1] = out[2] = out[3] = 0;
189 for (std::size_t i = 0; i < count; ++i) {
190 std::uint64_t mask = eq_mask(static_cast<std::uint64_t>(i),
191 static_cast<std::uint64_t>(index));
192 out[0] |= table[i][0] & mask;
193 out[1] |= table[i][1] & mask;
194 out[2] |= table[i][2] & mask;
195 out[3] |= table[i][3] & mask;
196 }
