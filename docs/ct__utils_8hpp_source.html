35// -- Byte-level Constant-Time Compare -----------------------------------------
36// Returns true if a[0..len) == b[0..len). Constant-time (no early exit).
37inline bool ct_equal(const void* a, const void* b, std::size_t len) noexcept {
38 const auto* pa = static_cast<const std::uint8_t*>(a);
39 const auto* pb = static_cast<const std::uint8_t*>(b);
40
41 std::uint64_t diff = 0;
42 // Process 8 bytes at a time
43 std::size_t i = 0;
44 for (; i + 8 <= len; i += 8) {
45 std::uint64_t va, vb;
46 std::memcpy(&va, pa + i, 8);
47 std::memcpy(&vb, pb + i, 8);
48 diff |= va ^ vb;
49 }
50 // Remaining bytes
51 for (; i < len; ++i) {
52 diff |= static_cast<std::uint64_t>(pa[i] ^ pb[i]);
53 }
54
55 return is_zero_mask(diff) != 0;
56}

65// -- Byte-level Constant-Time Conditional Copy --------------------------------
66// if (flag) memcpy(dst, src, len); -- constant-time
67inline void ct_memcpy_if(void* dst, const void* src, std::size_t len,
68 bool flag) noexcept {
69 auto mask = bool_to_mask(flag);
70 auto mask8 = static_cast<std::uint8_t>(mask & 0xFF);
71
72 auto* pd = static_cast<std::uint8_t*>(dst);
73 const auto* ps = static_cast<const std::uint8_t*>(src);
74
75 for (std::size_t i = 0; i < len; ++i) {
76 pd[i] ^= (pd[i] ^ ps[i]) & mask8;
77 }
78}

80// -- Constant-Time Conditional Swap -------------------------------------------
81// if (flag) swap(a[0..len), b[0..len)); -- constant-time
82inline void ct_memswap_if(void* a, void* b, std::size_t len,
83 bool flag) noexcept {
84 auto mask = bool_to_mask(flag);
85 auto mask8 = static_cast<std::uint8_t>(mask & 0xFF);
86
87 auto* pa = static_cast<std::uint8_t*>(a);
88 auto* pb = static_cast<std::uint8_t*>(b);
89
90 for (std::size_t i = 0; i < len; ++i) {
91 std::uint8_t diff = (pa[i] ^ pb[i]) & mask8;
92 pa[i] ^= diff;
93 pb[i] ^= diff;
94 }
95}

97// -- Constant-Time Zero Check -------------------------------------------------
98// Returns true if all bytes are zero. Constant-time.
99inline bool ct_is_zero(const void* data, std::size_t len) noexcept {
100 const auto* p = static_cast<const std::uint8_t*>(data);
101 std::uint64_t acc = 0;
102 for (std::size_t i = 0; i < len; ++i) {
103 acc |= static_cast<std::uint64_t>(p[i]);
104 }
105 return is_zero_mask(acc) != 0;
106}

125// -- Constant-Time Byte Select ------------------------------------------------
126// Returns a if flag is true, b otherwise. No branch.
127inline std::uint8_t ct_select_byte(std::uint8_t a, std::uint8_t b,
128 bool flag) noexcept {
129 auto mask = bool_to_mask(flag);
130 auto mask8 = static_cast<std::uint8_t>(mask & 0xFF);
131 return static_cast<std::uint8_t>((a & mask8) | (b & ~mask8));

134// -- Constant-Time Lexicographic Compare --------------------------------------
135// Returns: -1 if a < b, 0 if a == b, 1 if a > b. Fully branchless.
136//
137// For the common 32-byte case: fully unrolled, no loops, no loop-carried
138// dependencies. This prevents the compiler from inserting branches on
139// the "decided" flag to short-circuit iterations.

143// Branchless unsigned compare: returns {gt, lt} where each is 0 or 1.
144// value_barrier on BOTH inputs prevents Clang from inserting beq/bne
145// branches before the sltu instructions (observed on RISC-V Clang 21).
146inline void ct_cmp_pair(std::uint64_t wa, std::uint64_t wb,
147 std::uint64_t& gt, std::uint64_t& lt) noexcept {
148 ct::value_barrier(wa);
149 ct::value_barrier(wb);
150#if defined(__riscv) && (__riscv_xlen == 64)
151 asm volatile("sltu %0, %2, %1" : "=r"(gt) : "r"(wa), "r"(wb));
152 asm volatile("sltu %0, %2, %1" : "=r"(lt) : "r"(wb), "r"(wa));
153#else
154 gt = static_cast<std::uint64_t>(wa > wb);
155 lt = static_cast<std::uint64_t>(wa < wb);
156#endif
157}

159// Load 8 bytes + bswap for lexicographic order.
160// On RISC-V: may_alias avoids GCC decomposing memcpy into 8x lbu + sb chain.
161// Callers MUST pass 8-byte-aligned pointers (true for all ct_compare paths:
162// hash outputs, key data, heap-allocated test buffers are always aligned).
163inline std::uint64_t ct_load_be(const std::uint8_t* p) noexcept {
164#if defined(__riscv) && (__riscv_xlen == 64)
165 typedef std::uint64_t u64_alias __attribute__((__may_alias__));
166 std::uint64_t v = *reinterpret_cast<const u64_alias*>(p);
167 return __builtin_bswap64(v);
168#else
169 std::uint64_t v;
170 std::memcpy(&v, p, 8);
171#if defined(__GNUC__) || defined(__clang__)
172 return __builtin_bswap64(v);
173#elif defined(_MSC_VER)
174 return _byteswap_uint64(v);
175#else
176 // Generic fallback
177 return ((v >> 56) & 0xFF) | ((v >> 40) & 0xFF00) |
178 ((v >> 24) & 0xFF0000) | ((v >> 8) & 0xFF000000) |
179 ((v << 8) & 0xFF00000000) | ((v << 24) & 0xFF0000000000) |
180 ((v << 40) & 0xFF000000000000) | ((v << 56));
181#endif
182#endif
183}

187inline int ct_compare(const void* a, const void* b, std::size_t len) noexcept {
188 const auto* pa = static_cast<const std::uint8_t*>(a);
189 const auto* pb = static_cast<const std::uint8_t*>(b);
190
191 // ---- Fast path: 32 bytes (fully unrolled, zero branches) ----
192 // Algorithm: reverse-scan accumulation.
193 // Process words 3->2->1->0 (least significant first).
194 // Each differing word OVERRIDES the running result.
195 // Final result reflects the FIRST (most significant) differing word.
196 // value_barrier after every step prevents Clang from injecting
197 // beq/bne branches (observed with Clang 21 RISC-V).
198 if (len == 32) {
199 using namespace ct_compare_detail;
200
201 // Load all 4 word pairs in big-endian (lexicographic order)
202 std::uint64_t w0a = ct_load_be(pa + 0), w0b = ct_load_be(pb + 0);
203 std::uint64_t w1a = ct_load_be(pa + 8), w1b = ct_load_be(pb + 8);
204 std::uint64_t w2a = ct_load_be(pa + 16), w2b = ct_load_be(pb + 16);
205 std::uint64_t w3a = ct_load_be(pa + 24), w3b = ct_load_be(pb + 24);
206
207 std::uint64_t result = 0;
208
209 // Word 3 (bytes 24-31, least significant)
210 {
211 std::uint64_t gt, lt;
212 ct_cmp_pair(w3a, w3b, gt, lt);
213 std::uint64_t differs = gt | lt; // 0 or 1
214 ct::value_barrier(differs);
215 std::uint64_t mask = 0ULL - differs;
216 ct::value_barrier(mask);
217 result = (gt - lt) & mask; // result was 0
218 }
219 ct::value_barrier(result);
220
221 // Word 2 (bytes 16-23)
222 {
223 std::uint64_t gt, lt;
224 ct_cmp_pair(w2a, w2b, gt, lt);
225 std::uint64_t differs = gt | lt;
226 ct::value_barrier(differs);
227 std::uint64_t mask = 0ULL - differs;
228 ct::value_barrier(mask);
229 result = ((gt - lt) & mask) | (result & ~mask);
230 }
231 ct::value_barrier(result);
232
233 // Word 1 (bytes 8-15)
234 {
235 std::uint64_t gt, lt;
236 ct_cmp_pair(w1a, w1b, gt, lt);
237 std::uint64_t differs = gt | lt;
238 ct::value_barrier(differs);
239 std::uint64_t mask = 0ULL - differs;
240 ct::value_barrier(mask);
241 result = ((gt - lt) & mask) | (result & ~mask);
242 }
243 ct::value_barrier(result);
244
245 // Word 0 (bytes 0-7, most significant -- overrides all)
246 {
247 std::uint64_t gt, lt;
248 ct_cmp_pair(w0a, w0b, gt, lt);
249 std::uint64_t differs = gt | lt;
250 ct::value_barrier(differs);
251 std::uint64_t mask = 0ULL - differs;
252 ct::value_barrier(mask);
253 result = ((gt - lt) & mask) | (result & ~mask);
254 }
255
256#if defined(__GNUC__) || defined(__clang__)
257 asm volatile("" : "+r"(result));
258#endif
259 return static_cast<int>(static_cast<std::int64_t>(result));
260 }
261
262 // ---- General path: arbitrary length ----
263 std::uint64_t result = 0;
264 std::uint64_t decided = 0;
265
266 std::size_t i = 0;
267 for (; i + 8 <= len; i += 8) {
268 std::uint64_t wa = ct_compare_detail::ct_load_be(pa + i);
269 std::uint64_t wb = ct_compare_detail::ct_load_be(pb + i);
270 // Barrier inputs: prevent compiler from comparing wa/wb directly
271 // (Clang 21 RISC-V inserts beq before sltu without these)
272 ct::value_barrier(wa);
273 ct::value_barrier(wb);
274 std::uint64_t xor_val = wa ^ wb;
275 // nz = 1 if words differ, 0 otherwise
276#if defined(__riscv) && (__riscv_xlen == 64)
277 std::uint64_t nz;
278 asm volatile("snez %0, %1" : "=r"(nz) : "r"(xor_val));
279#else
280 std::uint64_t nz = ((xor_val | (0ULL - xor_val)) >> 63) & 1ULL;
281#endif
282
283 // Barrier on decided only: prevent compiler from short-circuiting
284 ct::value_barrier(decided);
285
286 // take = 1 only for the very first differing word
287 std::uint64_t take = nz & (1ULL - decided);
288
289 // mask = all-ones when take==1, zero when take==0
290 std::uint64_t mask = 0ULL - take;
291 ct::value_barrier(mask);
292
293 // Branchless unsigned compare: ct_cmp_pair-style barriers on inputs
294 std::uint64_t gt, lt;
295 ct::value_barrier(wa);
296 ct::value_barrier(wb);
297#if defined(__riscv) && (__riscv_xlen == 64)
298 asm volatile("sltu %0, %2, %1" : "=r"(gt) : "r"(wa), "r"(wb));
299 asm volatile("sltu %0, %2, %1" : "=r"(lt) : "r"(wb), "r"(wa));
300#else
301 gt = static_cast<std::uint64_t>(wa > wb);
302 lt = static_cast<std::uint64_t>(wa < wb);
303#endif
304 // diff_sign encodes: 1 = a>b, 0 = equal, -1 (0xFFFF...) = a<b
305 std::uint64_t diff_sign = gt - lt;
306
307 result = (diff_sign & mask) | (result & ~mask);
308 decided |= nz;
309
310#if defined(__GNUC__) || defined(__clang__)
311 asm volatile("" : "+r"(result), "+r"(decided));
312#endif
313 }
314
315 // Remaining bytes (< 8) -- byte-by-byte fallback
316 for (; i < len; ++i) {
317 std::uint64_t ai = pa[i];
318 std::uint64_t bi = pb[i];
319 std::uint64_t diff = ai ^ bi;
320
321#if defined(__riscv) && (__riscv_xlen == 64)
322 std::uint64_t nz;
323 asm volatile("snez %0, %1" : "=r"(nz) : "r"(diff));
324#else
325 std::uint64_t nz = ((diff | (0ULL - diff)) >> 63) & 1ULL;
326#endif
327 ct::value_barrier(decided);
328 std::uint64_t take = nz & (1ULL - decided);
329 std::uint64_t mask = 0ULL - take;
330 ct::value_barrier(mask);
331
332 std::uint64_t gt_b, lt_b;
333 ct::value_barrier(ai);
334 ct::value_barrier(bi);
335#if defined(__riscv) && (__riscv_xlen == 64)
336 asm volatile("sltu %0, %2, %1" : "=r"(gt_b) : "r"(ai), "r"(bi));
337 asm volatile("sltu %0, %2, %1" : "=r"(lt_b) : "r"(bi), "r"(ai));
338#else
339 gt_b = static_cast<std::uint64_t>(ai > bi);
340 lt_b = static_cast<std::uint64_t>(ai < bi);
341#endif
342 std::uint64_t diff_sign = gt_b - lt_b;
343
344 result = (diff_sign & mask) | (result & ~mask);
345 decided |= nz;
346
347#if defined(__GNUC__) || defined(__clang__)
348 asm volatile("" : "+r"(result), "+r"(decided));
349#endif
350 }
351
352 // Normalise to {-1, 0, 1} without branches.
353 // result is 0, 1, or 0xFFFFFFFFFFFFFFFF (-1 as uint64)
354 return static_cast<int>(static_cast<std::int64_t>(result));
355}
