8// Wraps BIP-32 derivation with coin-aware path construction:
9// m / purpose' / coin_type' / account' / change / address_index

18// auto master = bip32_master_key(seed, 64);

22// // Ethereum at m/44'/60'/0'/0/0
23// auto eth_key = coin_derive_key(master.first, Ethereum, 0, false, 0);
