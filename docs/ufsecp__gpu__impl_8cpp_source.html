26/* Hard upper bound on user-supplied GPU batch counts.
27 * Prevents hostile callers from triggering multi-GB allocations. */
28static constexpr std::size_t kMaxGpuBatchN = std::size_t{1} << 26; /* 64 M */

31#define UFSECP_GPU_CATCH \
32 catch (const std::bad_alloc&) { return UFSECP_ERR_GPU_MEMORY; } \
33 catch (...) { return UFSECP_ERR_INTERNAL; }

89ufsecp_error_t ufsecp_gpu_device_info(
90 uint32_t bid, uint32_t device_index,
91 ufsecp_gpu_device_info_t* info_out)
92{
93 if (!info_out) return UFSECP_ERR_NULL_ARG;
94 try {
95 auto b = create_backend(bid);
96 if (!b) return UFSECP_ERR_GPU_UNAVAILABLE;
97
98 DeviceInfo di;
99 auto err = b->device_info(device_index, di);
100 if (err != GpuError::Ok) return to_abi_error(err);
101
102 std::memcpy(info_out->name, di.name, sizeof(info_out->name));
103 info_out->global_mem_bytes = di.global_mem_bytes;
104 info_out->compute_units = di.compute_units;
105 info_out->max_clock_mhz = di.max_clock_mhz;
106 info_out->max_threads_per_block = di.max_threads_per_block;
107 info_out->backend_id = di.backend_id;
108 info_out->device_index = di.device_index;
109 return UFSECP_OK;
110 } UFSECP_GPU_CATCH
111}

117ufsecp_error_t ufsecp_gpu_ctx_create(
118 ufsecp_gpu_ctx** ctx_out,
119 uint32_t bid,
120 uint32_t device_index)
121{
122 if (!ctx_out) return UFSECP_ERR_NULL_ARG;
123 *ctx_out = nullptr;
124 try {
125 auto backend = create_backend(bid);
126 if (!backend) return UFSECP_ERR_GPU_UNAVAILABLE;
127
128 auto err = backend->init(device_index);
129 if (err != GpuError::Ok) return to_abi_error(err);
130
131 auto* ctx = new (std::nothrow) ufsecp_gpu_ctx;
132 if (!ctx) return UFSECP_ERR_INTERNAL;
133
134 ctx->backend = std::move(backend);
135 ctx->backend_id = bid;
136 ctx->device_index = device_index;
137 *ctx_out = ctx;
138 return UFSECP_OK;
139 } UFSECP_GPU_CATCH
140}
