73ufsecp_error_t ufsecp_gpu_device_info(
74 uint32_t bid, uint32_t device_index,
75 ufsecp_gpu_device_info_t* info_out)
76{
77 if (!info_out) return UFSECP_ERR_NULL_ARG;
78
79 auto b = create_backend(bid);
80 if (!b) return UFSECP_ERR_GPU_UNAVAILABLE;
81
82 DeviceInfo di;
83 auto err = b->device_info(device_index, di);
84 if (err != GpuError::Ok) return to_abi_error(err);
85
86 std::memcpy(info_out->name, di.name, sizeof(info_out->name));
87 info_out->global_mem_bytes = di.global_mem_bytes;
88 info_out->compute_units = di.compute_units;
89 info_out->max_clock_mhz = di.max_clock_mhz;
90 info_out->max_threads_per_block = di.max_threads_per_block;
91 info_out->backend_id = di.backend_id;
92 info_out->device_index = di.device_index;
93 return UFSECP_OK;
94}

100ufsecp_error_t ufsecp_gpu_ctx_create(
101 ufsecp_gpu_ctx** ctx_out,
102 uint32_t bid,
103 uint32_t device_index)
104{
105 if (!ctx_out) return UFSECP_ERR_NULL_ARG;
106 *ctx_out = nullptr;
107
108 auto backend = create_backend(bid);
109 if (!backend) return UFSECP_ERR_GPU_UNAVAILABLE;
110
111 auto err = backend->init(device_index);
112 if (err != GpuError::Ok) return to_abi_error(err);
113
114 auto* ctx = new (std::nothrow) ufsecp_gpu_ctx;
115 if (!ctx) return UFSECP_ERR_INTERNAL;
116
117 ctx->backend = std::move(backend);
118 ctx->backend_id = bid;
119 ctx->device_index = device_index;
120 *ctx_out = ctx;
121 return UFSECP_OK;
122}
