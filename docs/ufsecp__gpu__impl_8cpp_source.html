30/* Hard upper bound on user-supplied GPU batch counts.
31 * Prevents hostile callers from triggering multi-GB allocations. */
32static constexpr std::size_t kMaxGpuBatchN = std::size_t{1} << 26; /* 64 M */

57static bool has_valid_compressed_pubkeys(const uint8_t* pubkeys33, size_t count) {
58 for (size_t index = 0; index < count; ++index) {
59 const uint8_t prefix = pubkeys33[index * 33];
60 if (prefix != 0x02 && prefix != 0x03) {
61 return false;
62 }
63 }
64 return true;
65}
66
67static bool has_valid_uncompressed_pubkeys(const uint8_t* pubkeys65, size_t count) {
68 for (size_t index = 0; index < count; ++index) {
69 if (pubkeys65[index * 65] != 0x04) {
70 return false;
71 }
72 }
73 return true;
74}
75
76static bool has_valid_recovery_ids(const int* recids, size_t count) {
77 for (size_t index = 0; index < count; ++index) {
78 if (recids[index] < 0 || recids[index] > 3) {
79 return false;
80 }
81 }
82 return true;
83}

85static bool has_valid_bip324_sizes(const uint32_t* sizes, size_t count, uint32_t max_payload) {
86 for (size_t index = 0; index < count; ++index) {
87 if (sizes[index] > max_payload) {
88 return false;
89 }
90 }
91 return true;
92}

94static bool has_valid_bulletproof_prefixes(const uint8_t* proofs324, size_t count) {
95 for (size_t index = 0; index < count; ++index) {
96 const uint8_t* proof = proofs324 + (index * 324);
97 if (proof[0] != 0x04 || proof[65] != 0x04 || proof[130] != 0x04 || proof[195] != 0x04) {
98 return false;
99 }
100 }
101 return true;
102}

168ufsecp_error_t ufsecp_gpu_ctx_create(
169 ufsecp_gpu_ctx** ctx_out,
170 uint32_t bid,
171 uint32_t device_index)
172{
173 if (!ctx_out) return UFSECP_ERR_NULL_ARG;
174 *ctx_out = nullptr;
175 try {
176 auto backend = create_backend(bid);
177 if (!backend) return UFSECP_ERR_GPU_UNAVAILABLE;
178
179 auto err = backend->init(device_index);
180 if (err != GpuError::Ok) return to_abi_error(err);
181
182 auto* ctx = new (std::nothrow) ufsecp_gpu_ctx;
183 if (!ctx) return UFSECP_ERR_INTERNAL;
184
185 ctx->backend = std::move(backend);
186 ctx->backend_id = bid;
187 ctx->device_index = device_index;
188 *ctx_out = ctx;
189 return UFSECP_OK;
190 } UFSECP_GPU_CATCH
191}

565/* ===========================================================================
566 * BIP-352 scan plan precomputation (CPU utility, no GPU ctx needed)
567 * =========================================================================== */
568
569ufsecp_error_t ufsecp_bip352_prepare_scan_plan(
570 const uint8_t scan_privkey32[32],
571 uint8_t plan264_out[264])
572{
573 if (!scan_privkey32 || !plan264_out) return UFSECP_ERR_NULL_ARG;
574
575 using namespace secp256k1::fast;
576
577 Scalar k = Scalar::from_bytes(scan_privkey32);
578 if (k.is_zero()) return UFSECP_ERR_BAD_KEY;
579
580 auto decomp = glv_decompose(k);
581 auto k1_bytes = decomp.k1.to_bytes();
582 auto k2_bytes = decomp.k2.to_bytes();
583
584 /* Layout (264 bytes, matches OpenCL BIP352ScanKeyGlv):
585 * [0..129] wnaf1[130] — wNAF digits for k1 half-scalar
586 * [130..259] wnaf2[130] — wNAF digits for k2 half-scalar
587 * [260] k1_neg — 1 if k1 was negative
588 * [261] flip_phi — 1 if phi table y should be negated
589 * [262..263] pad — zero padding */
590 auto* wn1 = reinterpret_cast<int8_t*>(plan264_out);
591 auto* wn2 = reinterpret_cast<int8_t*>(plan264_out + 130);
592 plan264_out[260] = decomp.k1_neg ? 1 : 0;
593 plan264_out[261] = (decomp.k1_neg != decomp.k2_neg) ? 1 : 0;
594 plan264_out[262] = 0;
595 plan264_out[263] = 0;
596
597 /* 5-bit wNAF encoding — mirrors host_compute_wnaf() used by the
598 * bip352_pipeline_kernel OpenCL kernel. */
599 auto compute_wnaf = [](const uint8_t* be32, int8_t out[130]) {
600 uint64_t s[4] = {};
601 for (int limb = 0; limb < 4; ++limb) {
602 uint64_t v = 0;
603 int base = limb * 8;
604 for (int i = 0; i < 8; ++i) v = (v << 8) | be32[base + i];
605 s[3 - limb] = v;
606 }
607 for (int i = 0; i < 130; ++i) {
608 if (s[0] & 1ULL) {
609 int d = (int)(s[0] & 0x1FULL);
610 if (d >= 16) {
611 d -= 32;
612 uint64_t add = static_cast<uint64_t>(-d);
613 uint64_t prev = s[0]; s[0] += add;
614 if (s[0] < prev) { for (int j = 1; j < 4; ++j) if (++s[j]) break; }
615 } else {
616 uint64_t prev = s[0]; s[0] -= static_cast<uint64_t>(d);
617 if (s[0] > prev) { for (int j = 1; j < 4; ++j) if (s[j]--) break; }
618 }
619 out[i] = static_cast<int8_t>(d);
620 } else {
621 out[i] = 0;
622 }
623 s[0] = (s[0] >> 1) | (s[1] << 63);
624 s[1] = (s[1] >> 1) | (s[2] << 63);
625 s[2] = (s[2] >> 1) | (s[3] << 63);
626 s[3] >>= 1;
627 }
628 };
629
630 compute_wnaf(k1_bytes.data(), wn1);
631 compute_wnaf(k2_bytes.data(), wn2);
632 return UFSECP_OK;
633}
